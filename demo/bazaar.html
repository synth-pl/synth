// let infer — model resolves type
fn apply_filters(
  state: AppState
) -> Item[] {
  let infer a = by_category(
    state.all_items,
    state.filter_cat
  )
  let infer b = by_rarity(a, state.filter_rar)
  let infer c = search_items(b, state.search)
  let infer sorted =
    apply_sort(c, state.sort_key)
  sorted
}
