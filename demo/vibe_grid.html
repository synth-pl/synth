// VIBE GRID — Synth idle game

store Game {
  vibes:       float = 0
  total_vibes: float = 0
  vps:         float = 0
  counts:      any   = [0,0,0,0,0,0,0]
}

// @memo caches results
@memo
fn building_cost(base_cost, owned) =
  floor(base_cost * pow(1.15, owned))

fn buy(idx) {
  let b    = catalog[idx]
  let cost = building_cost(b.base_cost,
                 Game.counts[idx])
  if Game.vibes >= cost {
    let new_counts = set_at(Game.counts,
                    idx, Game.counts[idx]+1)
    let new_vps = catalog.sum_by(
                    b => b.rate * new_counts[b.id])
    Game.set({ vibes: Game.vibes - cost,
               vps: new_vps,
               counts: new_counts })
  }
}

// do notation: scoped computation
fn render_building(b) {
  let info = do {
    let owned = Game.counts[b.id]
    let cost  = building_cost(b.base_cost, owned)
    let cls   = Game.vibes >= cost ? "can-afford" : ""
    { owned, cost, cls }
  }
  "<div class=..." + info.cls + "..."
}

// Pipeline showcase
fn total_owned() =
  Game.counts
    |> fold(0, (acc, n) => acc + n)

on Game.change { render() }
